    import random
    import sys
    # Define the possible moves
    MOVES = ["rock", "paper", "scissors"]
    # Initialize the history tracker for the AI
    # This dictionary tracks what the user played (inner key) 
    # immediately after playing a previous move (outer key).
    # Example: history["rock"]["paper"] is the count of times the user played 'paper' 
    # right after playing 'rock'.
    history = {
        "rock": {"rock": 0, "paper": 0, "scissors": 0},
        "paper": {"rock": 0, "paper": 0, "scissors": 0},
        "scissors": {"rock": 0, "paper": 0, "scissors": 0},
    }
    # Variable to hold the user's move from the previous round
    last_user_move = None 
    # Score tracking
    scores = {"player": 0, "computer": 0, "tie": 0}
    def get_counter(move):
        """Returns the move that beats the given move."""
        if move == "rock":
            return "paper"
        elif move == "paper":
            return "scissors"
        elif move == "scissors":
            return "rock"
        return random.choice(MOVES) # Default fallback
    def get_computer_choice(user_move_history):
        """
        The Advanced AI logic.
        Predicts the user's next move based on the last move played,
        then chooses the counter-move.
        """
        global last_user_move
        # 1. Prediction Logic: Only try to predict if we have at least one round of history
        if last_user_move is not None:
            
            # Get the transition counts for the last move the user played
            transition_counts = history[last_user_move]
            
            # Check if the AI has seen any follow-up moves after the last move
            if sum(transition_counts.values()) > 0:
                
                # Find the move the user plays most frequently after 'last_user_move'
                predicted_move = max(transition_counts, key=transition_counts.get)
                
                # Check if this move has actually been played more than 0 times
                if transition_counts[predicted_move] > 0:
                    # 2. Counter Strategy: Choose the move that beats the predicted move
                    ai_choice = get_counter(predicted_move)
                    print(f"\n AI analyzed your pattern and predicted you would play {predicted_move.upper()}.")
                    return ai_choice
        
        # Fallback: If no history or pattern is found, play randomly
        return random.choice(MOVES)
    def determine_winner(player, computer):
        """Compares moves and updates scores."""
        global scores
        
        if player == computer:
            scores["tie"] += 1
            return "It's a TIE!"
        
        # Winning Logic: (Rock beats Scissors, Scissors beats Paper, Paper beats Rock)
        elif (player == "rock" and computer == "scissors") or \
             (player == "scissors" and computer == "paper") or \
             (player == "paper" and computer == "rock"):
            scores["player"] += 1
            return " YOU WIN! "
        else:
            scores["computer"] += 1
            return " COMPUTER WINS! "
    def play_game():
        """Main function to run the game loop."""
        global last_user_move
        
        print("--- Rock, Paper, Scissors with Learning AI ---")
        print("The AI will try to learn your patterns to beat you!")
        while True:
            try:
                # --- 1. Get AI's Choice ---
                # The AI makes its choice before asking the user for input
                computer_choice = get_computer_choice(history)
                
                # --- 2. Get User Input ---
                user_input = input("\nEnter your choice (R)ock, (P)aper, (S)cissors, or (Q)uit: ").lower()
                
                if user_input in ('q', 'quit'):
                    print("\nThanks for playing! Final Scores:")
                    print(f"Player Wins: {scores['player']}, AI Wins: {scores['computer']}, Ties: {scores['tie']}")
                    sys.exit(0)
                # Map single letters to full moves
                if user_input == 'r':
                    user_choice = "rock"
                elif user_input == 'p':
                    user_choice = "paper"
                elif user_input == 's':
                    user_choice = "scissors"
                else:
                    print(" Invalid input. Please enter R, P, S, or Q.")
                    continue
                # --- 3. Display Choices & Determine Winner ---
                print(f"You chose: {user_choice.upper()}")
                print(f"AI chose: {computer_choice.upper()}")
                
                result = determine_winner(user_choice, computer_choice)
                print(f"\n--- {result} ---")
                print(f"Current Score: You {scores['player']} - AI {scores['computer']} - Tie {scores['tie']}")
                # --- 4. Update AI History (The Learning Step) ---
                if last_user_move is not None:
                    # Increment the count for the transition: last_user_move -> user_choice
                    history[last_user_move][user_choice] += 1
                    
                # Set the current user move as the 'last_user_move' for the next round
                last_user_move = user_choice
                
            except KeyboardInterrupt:
                print("\nGame interrupted. Exiting.")
                sys.exit(0)
            except Exception as e:
                print(f"An unexpected error occurred: {e}")
                break
    if __name__ == "__main__":
        play_game()
